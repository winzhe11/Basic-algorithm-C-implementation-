#### 1.1.2 整数二分，浮点数二分

*2024年4月15日*

**整数二分模板**

```c++
//要找的答案在左边
int bsearch_1(int l, int r) {
    while (l < r) {
        int mid = l + r >> 1;
        if (check(mid)) r = mid; //check()判断mid是否满足性质
        else l = mid + 1;
    }
    return l;
}
//要找的答案在右边
int bsearch_2(int l, int r) {
    while (l < r) {
        int mid = l + r + 1>> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}
```

**浮点数二分**

```c++
#include <iostream>
using namespace std;
int main() {
    int a = 2;
    double l = 0, r = a;
    while (r - l > 1e-15) {
        double mid = (l + r) / 2;
        if (mid * mid >= a) r = mid;
        else l = mid;
    }
    cout << l;
    return 0;
}
```
