## 2.3.2 字符串哈希

**不多说， 安静的享受神作吧！！！**

```c++
#include <iostream>

using namespace std;
const int N = 1e5 + 10, P = 131;
char str[N];
unsigned long long p[N], h[N];

unsigned long long find(int l, int r) {
    return h[r] - h[l - 1] * p[r - l + 1];
}

int main() {
    int n, m;
    cin >> str + 1 >> n >> m;
    p[0] = 1;
    for (int i = 1; i <= n; i++) {
        p[i] = p[i - 1] * P;
        h[i] = h[i - 1] * P + str[i];
    }
    while (m--) {
        int l1, r1, l2, r2;
        cin >> l1 >> r1 >> l2 >> r2;
        if (find(l1, r1) == find(l2, r2)) puts("Yes");
        else puts("No");
    }
    return 0;
}
```
